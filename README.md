# DWUID

## Install
`npm i --save @dwlib/dwuid`

## Usage
```javascript
// CJS
const DWUID = require('@dwlib/dwuid');
const DWUIDCreate = require('@dwlib/dwuid/create');
const DWUIDDecode = require('@dwlib/dwuid/decode');
const DWUIDGenerate = require('@dwlib/dwuid/generate');
const DWUIDGenerator = require('@dwlib/dwuid/generator');
const DWUIDIsValid = require('@dwlib/dwuid/isValid');
const DWUIDNormalize = require('@dwlib/dwuid/normalize');
const DWUIDTimestamp = require('@dwlib/dwuid/timestamp');
const DWUIDValidate = require('@dwlib/dwuid/validate');
// ESM
import DWUID from '@dwlib/dwuid';
import DWUIDCreate from '@dwlib/dwuid/create';
import DWUIDDecode from '@dwlib/dwuid/decode';
import DWUIDGenerate from '@dwlib/dwuid/generate';
import DWUIDGenerator from '@dwlib/dwuid/generator';
import DWUIDIsValid from '@dwlib/dwuid/isValid';
import DWUIDNormalize from '@dwlib/dwuid/normalize';
import DWUIDTimestamp from '@dwlib/dwuid/timestamp';
import DWUIDValidate from '@dwlib/dwuid/validate';
```
